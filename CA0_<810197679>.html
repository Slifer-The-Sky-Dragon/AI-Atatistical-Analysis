import math
sum_gas = 0
cnt_gas = 0
sum_dis = 0
cnt_dis = 0

start_it = time.time()

for index, row in df.iterrows():
    if row['fueltype'] == 0 and math.isnan(row['price']) == False:
        cnt_gas += 1
        sum_gas += row['price']
    elif row['fueltype'] == 1 and math.isnan(row['price']) == False:
        cnt_dis += 1
        sum_dis += row['price']
print('gas Mean = ' + str(sum_gas / cnt_gas))
print('diesel Mean = ' +  str(sum_dis / cnt_dis))

end_it = time.time()

با توجه به روش normal equation می‌توان ضرایب را با رابطه زیر به دست آورد:

$\theta = (X ^ {T} X) ^ {-1} . (X ^ {T} Y)$

theta = pandas.DataFrame(numpy.linalg.pinv(theta.values), theta.columns, theta.index)
theta = theta.dot(X.T)

خط قرمز نشان دهنده‌ی تخمین خطی ما از رابطه price با curbweight است. خط به دست آمده کمترین خطا را نسبت به سایر خط های ممکن دارد و معادله‌ی آن برابر است با:

$price = 0.001578 \times curbweight - 4.035820$ , $\theta _{0} = -4.035820$ , $\theta _{1} = 0.001578$

با اینکه این خط رابطه بین price و curbweight را دقیقا نشان نمی‌دهد اما یک تقریب خطی مناسب از آن را در اختیار ما می‌گذارد.
